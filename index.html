3. Joins & Relationships

Combining data from transactions and customers.

[Image of SQL Joins Venn Diagram] 
-- Inner Join (Matches in both)
SELECT transaction_id, amount, first_name 
FROM transactions INNER JOIN customers 
ON transactions.customer_id = customers.customer_id;

-- Left Join (All transactions, even without customers)
SELECT * FROM transactions LEFT JOIN customers ON ...;

4. Advanced Topics New

Topics beyond basic CRUD operations:

A. Common Table Expressions (CTEs)

-- Improve readability of complex subqueries
WITH HourlyPayCTE AS (
    SELECT *, (hourly_pay * 2080) as annual_salary 
    FROM employees
)
SELECT * FROM HourlyPayCTE WHERE annual_salary > 50000;

B. Window Functions

-- Rank employees by pay within their job category
SELECT first_name, job, hourly_pay,
RANK() OVER(PARTITION BY job ORDER BY hourly_pay DESC) as pay_rank
FROM employees;

C. JSON Data Handling

-- MySQL supports native JSON types
ALTER TABLE products ADD attributes JSON;
INSERT INTO products (attributes) VALUES ('{"color": "red", "size": "large"}');
SELECT attributes->>"$.color" FROM products;

5. Automation: Triggers & Procedures

-- Automatically update salary when hourly pay changes
CREATE TRIGGER before_hourly_pay_update
BEFORE UPDATE ON employees
FOR EACH ROW
SET NEW.salary = (NEW.hourly_pay * 2080);

-- Stored Procedure with Parameters
DELIMITER //
CREATE PROCEDURE find_customer(IN id INT)
BEGIN
    SELECT * FROM customers WHERE customer_id = id;
END //
DELIMITER ;
